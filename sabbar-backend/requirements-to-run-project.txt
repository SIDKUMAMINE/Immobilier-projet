"""
FastAPI Main Application - Configuration CORS pour Railway
"""

from fastapi import +
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import JSONResponse
import os
from dotenv import load_dotenv

# Charger les variables d'environnement
load_dotenv()

app = FastAPI(
    title="SABBAR API",
    description="API Immobilière avec Railway deployment",
    version="1.0.0"
)

# ============================================================================
# CORS MIDDLEWARE - Configuration pour Railway
# ============================================================================
app.add_middleware(
    CORSMiddleware,
    allow_origins=[
        # LOCAL
        "http://localhost:3000",
        "http://localhost:8000",
        "http://127.0.0.1:3000",
        "http://127.0.0.1:8000",
        # PRODUCTION - landmark-estate.com
        "https://landmark-estate.com",
        "https://www.landmark-estate.com",
        # PRODUCTION - Vercel
        "https://immobilier-projet-sexp.vercel.app",
        # WILDCARD pour Railway (accepte tous les domaines Railway)
        "*.railway.app",
    ],
    allow_credentials=True,
    allow_methods=["GET", "POST", "PUT", "DELETE", "OPTIONS"],
    allow_headers=["Content-Type", "Authorization"],
)

# ============================================================================
# HEALTH CHECK
# ============================================================================
@app.get("/health")
async def health():
    """Vérifier l'API"""
    return {"status": "ok", "service": "sabbar-api"}


@app.get("/health/db")
async def health_db():
    """Vérifier la connexion BD"""
    try:
        # Importer le client Supabase
        from app.db.supabase_client import get_db
        
        supabase = get_db()
        response = supabase.table("properties").select("count", count="exact").execute()
        return {"status": "ok", "database": "connected"}
    except Exception as e:
        return {"status": "error", "database": "disconnected", "error": str(e)}


@app.get("/")
async def root():
    """Endpoint racine"""
    return {
        "message": "SABBAR API - Railway Deployment",
        "version": "1.0.0",
        "docs": "/docs",
        "health": "/health"
    }


# ============================================================================
# INCLURE LES ROUTERS
# ============================================================================
# À adapter selon votre structure réelle
try:
    from app.api.v1.endpoints import auth as auth_routes
    
    app.include_router(
        auth_routes.router,
        prefix="/api/v1",
        tags=["authentication"]
    )
except ImportError:
    print("⚠️ Module auth non trouvé, en attente de configuration")


# ============================================================================
# LANCER L'APP
# ============================================================================
if __name__ == "__main__":
    import uvicorn
    port = int(os.getenv("PORT", 8000))
    uvicorn.run("main:app", host="0.0.0.0", port=port, reload=False)
